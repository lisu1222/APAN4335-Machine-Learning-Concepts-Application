my.sum <- function(x, the.type) {
  if(the.type == "even"){
    sum(x[(x %%2 == 0)], na.rm = TRUE)
  }
  else if(the.type == "odd"){
    sum(x[(x %%2 == 1)], na.rm = TRUE)
  }
  else if(the.type == "all"){
    sum(x, na.rm = TRUE)
  }
}
